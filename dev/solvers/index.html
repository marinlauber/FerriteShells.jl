Nonlinear analysis

Newton–Raphson method

u = zeros(n_dofs)
residual = compute_residual(u)
while norm(residual) > tol
    tangent_stiffness = compute_tangent_stiffness(u)
    du = tangent_stiffness \ residual
    u += du
    residual = compute_residual(u)
end
